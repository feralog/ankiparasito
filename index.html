async function startReview() { // Força update se versão for diferente if (localStorage.getItem('flashcardsVersion') !== 'v2') { localStorage.removeItem('flashcardsState'); localStorage.setItem('flashcardsVersion', 'v2'); } if (!loadState()) { const res = await fetch('parasito2.txt?v=2' + Date.now()); const text = await res.text(); parseText(text); current = 0; saveState(); } document.getElementById('controls').style.display = 'none'; document.getElementById('review').style.display = 'block'; showCard(); } document.getElementById('start-review').onclick = startReview;
